            # Create title
            title = Text("Awakening to Reality", font_size=48, color=harmonious_color("title"))
            title.move_to(UP * 3.8)
            self.add(title)
            polished_fade_in(title)

            # Create subtitle (derived from narration intro)
            subtitle = Text("Questioning the Nature of Reality", font_size=32, color=harmonious_color("subtitle"))
            subtitle.next_to(title, DOWN, buff=0.4)
            safe_position(subtitle)
            self.add(subtitle)
            polished_fade_in(subtitle)

            # BeatPlan setup
            num_beats = max(12, min(30, int(np.ceil(tracker.duration / 1.8))))
            bp = BeatPlan(num_beats, tracker.duration)

            # Central diagram: reality illusion (simple grid/mirror effect)
            grid = NumberPlane(x_range=(-3, 3, 1), y_range=(-2, 2, 1), background_line_style={"stroke_color": harmonious_color("grid"), "stroke_width": 1})
            grid.move_to(ORIGIN + DOWN * 0.6)
            self.add(grid)
            play_next(bp, FadeIn(grid), tracker)

            # Progressive bullets (from narration: "What is real?", "The choice awaits")
            bullets = VGroup()
            bullet1 = Text("• What is real?", font_size=28, color=harmonious_color("bullet")).set_max_width(6.0)
            bullet1.move_to(LEFT * 3.5 + UP * 1.5)
            bullets.add(bullet1)

            bullet2 = Text("• The choice awaits", font_size=28, color=harmonious_color("bullet")).set_max_width(6.0)
            bullet2.move_to(LEFT * 3.5 + UP * 0.5)
            bullets.add(bullet2)

            safe_layout(bullets)
            self.add(bullets)

            # Evolve right-side visual: blurred matrix code background, then sharpen to pill icon
            matrix_bg = Rectangle(width=4, height=6, color=harmonious_color("matrix_bg"), fill_opacity=0.5).move_to(RIGHT * 3 + ORIGIN)
            self.add(matrix_bg)
            play_text_next(bp, [bullet1], tracker, max_text_seconds=999)

            # Sharpen to show pill icon (red pill)
            red_pill = Circle(radius=0.5, color=RED, fill_opacity=1).move_to(matrix_bg.get_center())
            blue_pill = Circle(radius=0.5, color=BLUE, fill_opacity=1).move_to(matrix_bg.get_center() + LEFT * 1.2)
            pills = VGroup(red_pill, blue_pill)
            self.add(pills)
            play_text_next(bp, [bullet2], tracker, max_text_seconds=999)
            play_next(bp, Transform(matrix_bg, pills), tracker)  # Evolve the background to pills

            # Fade out elements as narration ends
            self.play(FadeOut(VGroup(title, subtitle, bullets, grid, matrix_bg, pills)), run_time=bp.get_remaining_time())
